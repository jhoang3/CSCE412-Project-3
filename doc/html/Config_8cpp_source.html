24bool parseIPRange(const std::string& s, uint32_t& outBase, int& outPrefix) {
25 std::istringstream iss(s);
26 int a, b, c, d;
27 char ch;
28 if (!(iss >> a >> ch && ch == '.' && iss >> b >> ch && ch == '.' && iss >> c >> ch && ch == '.' && iss >> d))
29 return false;
30 if (a < 0 || a > 255 || b < 0 || b > 255 || c < 0 || c > 255 || d < 0 || d > 255)
31 return false;
32 outBase = (static_cast<uint32_t>(a) << 24) | (static_cast<uint32_t>(b) << 16) |
33 (static_cast<uint32_t>(c) << 8) | static_cast<uint32_t>(d);
34 outPrefix = 32;
35 if (iss >> ch && ch == '/') {
36 if (!(iss >> outPrefix) || outPrefix < 0 || outPrefix > 32)
37 return false;
38 }
39 return true;
40}

52bool ipInRange(uint32_t ip, uint32_t base, int prefixLen) {
53 if (prefixLen <= 0) return true;
54 if (prefixLen >= 32) return ip == base;
55 uint32_t mask = static_cast<uint32_t>(0xFFFFFFFF) << (32 - prefixLen);
56 return (ip & mask) == (base & mask);
57}

61bool loadConfig(const std::string& path, Config& out) {
62 std::ifstream f(path);
63 if (!f) return false;
64
65 std::string line;
66 while (std::getline(f, line)) {
67 trim(line);
68 if (line.empty() || line[0] == '#') continue;
69
70 std::string::size_type eq = line.find('=');
71 if (eq == std::string::npos) continue;
72
73 std::string key = line.substr(0, eq);
74 std::string value = line.substr(eq + 1);
75 trim(key);
76 trim(value);
77
78 if (key == "scaleUpThreshold") {
79 int v = 0;
80 if (std::istringstream(value) >> v) out.scaleUpThreshold = v;
81 } else if (key == "scaleDownThreshold") {
82 int v = 0;
83 if (std::istringstream(value) >> v) out.scaleDownThreshold = v;
84 } else if (key == "cooldownCycles") {
85 int v = 0;
86 if (std::istringstream(value) >> v) out.cooldownCycles = v;
87 } else if (key == "initialQueueMultiplier") {
88 int v = 0;
89 if (std::istringstream(value) >> v) out.initialQueueMultiplier = v;
90 } else if (key == "taskTimeMin") {
91 int v = 1;
92 if (std::istringstream(value) >> v) out.taskTimeMin = v;
93 } else if (key == "taskTimeMax") {
94 int v = 100;
95 if (std::istringstream(value) >> v) out.taskTimeMax = v;
96 } else if (key == "blockedRange" || key == "blockedIP") {
97 if (!value.empty()) out.blockedIPRanges.push_back(value);
98 }
99 }
100 return true;
101}
