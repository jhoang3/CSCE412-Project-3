50void LoadBalancer::runCycle(std::ostream* log, bool addRandomRequest) {
51 ++system_clock;
52
53 int added_this_cycle = 0;
54 if (addRandomRequest && std::rand() % 2 == 0) {
55 Request r = generateRandomRequest(settings);
56 if (isIPBlocked(settings, r.IP_in)) {
57 ++total_dropped;
58 } else {
59 buffer.push(r);
60 added_this_cycle = 1;
61 }
62 }
63
64 int dispatched_this_cycle = 0;
65 int completed_this_cycle = 0;
66 /* Process each server: tick if busy (count completed jobs), else dispatch from queue if available. */
67 for (WebServer& server : cluster) {
68 if (server.isBusy()) {
69 if (server.tick()) {
70 ++total_processed;
71 ++completed_this_cycle;
72 }
73 } else if (!buffer.empty()) {
74 Request r = buffer.front();
75 buffer.pop();
76 server.dispatch(r);
77 ++dispatched_this_cycle;
78 }
79 }
80
81 const size_t servers_before = cluster.size();
82 scaleCluster();
83 const size_t servers_after = cluster.size();
84
85 telemetry(log, added_this_cycle, dispatched_this_cycle, completed_this_cycle);
86
87 if (log && servers_after != servers_before) {
88 if (servers_after > servers_before)
89 *log << " [scale up: " << servers_before << " -> " << servers_after << " servers]\n";
90 else
91 *log << " [scale down: " << servers_before << " -> " << servers_after << " servers]\n";
92 }
93}

95void LoadBalancer::scaleCluster() {
96 if (cooldown_timer > 0) {
97 --cooldown_timer;
98 return;
99 }
100 const size_t n = cluster.size();
101 const size_t q = buffer.size();
102 if (q > static_cast<size_t>(settings.scaleUpThreshold) * n) {
103 cluster.emplace_back();
104 ++scale_ups;
105 cooldown_timer = settings.cooldownCycles;
106 } else if (n > 1 && !cluster.back().isBusy() && q < static_cast<size_t>(settings.scaleDownThreshold) * n) {
107 cluster.pop_back();
108 ++scale_downs;
109 cooldown_timer = settings.cooldownCycles;
110 }
111}
