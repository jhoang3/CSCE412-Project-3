31void LoadBalancer::runCycle(std::ostream* log) {
32 ++system_clock;
33
34 /* Maybe add one new request this cycle (e.g. 50% chance). */
35 if (std::rand() % 2 == 0) {
36 Request r = generateRandomRequest();
37 if (isIPBlocked(settings, r.IP_in)) {
38 ++total_dropped;
39 } else {
40 buffer.push(r);
41 }
42 }
43
44 /* Process each server: tick if busy (count completed jobs), else dispatch from queue if available. */
45 for (WebServer& server : cluster) {
46 if (server.isBusy()) {
47 if (server.tick()) {
48 ++total_processed;
49 }
50 } else if (!buffer.empty()) {
51 Request r = buffer.front();
52 buffer.pop();
53 server.dispatch(r);
54 }
55 }
56
57 scaleCluster();
58 telemetry(log);
59}

61void LoadBalancer::scaleCluster() {
62 if (cooldown_timer > 0) {
63 --cooldown_timer;
64 return;
65 }
66 const size_t n = cluster.size();
67 const size_t q = buffer.size();
68 if (q > static_cast<size_t>(settings.scaleUpThreshold) * n) {
69 cluster.emplace_back();
70 ++scale_ups;
71 cooldown_timer = settings.cooldownCycles;
72 } else if (n > 1 && !cluster.back().isBusy() && q < static_cast<size_t>(settings.scaleDownThreshold) * n) {
73 cluster.pop_back();
74 ++scale_downs;
75 cooldown_timer = settings.cooldownCycles;
76 }
77}
